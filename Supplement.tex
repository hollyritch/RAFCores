\section{Glossary on Graph Theory}\label{sec:Glossary}

\paragraph{Graphs} A graph $G=(V,E)$ is a tuple composed of a set of vertices and 
a set of edges. Edges encode the adjacency of elements $u,v\in V$, i.e., $\{u,v\}\in E$ 
whenever $u$ is adjacent to $v$. We write $V(G)$ and $E(G)$ for the set of vertices 
and edges of a given graph $G$.
 
\paragraph{Directed Graphs} In a directed graph $G\coloneqq (V,E)$, edges are ordered 
pairs instead of unordered pairs, and $(u,v) \in E$ whenever there is an edge from $u$ to 
$v$. The underyling undirected graph of $G$ is the graph $G_{\circ} \coloneqq (V, E_{\circ})$ with
$E_{\circ}\coloneqq \{ \{u,v\} \ \vert \ (u,v) \in E(G) \}$.

\paragraph{Subgraphs} A subgraph of a (directed) graph $G$, denoted as $G'\subseteq G$ is a graph $G=(V', E')$ 
such that $V'\subseteq V$ and $E'\subseteq E$. A subgraph \emph{induced} by a set of vertices 
$V'\subseteq V$ is the graph $G[V']\coloneqq (V', F)$ with $F\coloneqq \{ \{u,v\} \ \vert \ u,v\in V', 
\{u,v\} \in E\}$ or in the directed case $F\coloneqq \{(u,v) \ \vert \ u,v\in V', (u,v) \in E\}$.

\paragraph{Paths and Directed Paths} A (directed) path $\mathcal{P}_{v_1v_k}$ in a (directed) graph $G$ is 
a sequence of pairwise distinct vertices $(v_1,\ldots, v_k)$ such that $\{v_i, v_{i+1}\}\in E(G) \  ((v_i, v_{i+1})\in E(G)), 
1\leq i\leq k-1$.

\paragraph{(Strongly) Connected (Sub-)Graphs} A subgraph $G'\coloneqq (V', E')$ of a (directed) graph 
$G=(V,E)$ is called (strongly) connected if for any two vertices $u,v\in V'$ there is a path 
$\mathcal{P}_{uv}$, which, in the directed case, implies that there is a path from $u$ to $v$ and 
from $v$ to $u$. If $G$ is directed, then $G'\subseteq G$ is called connected if $G'_\circ$ is connected. 
If $G'$ is not connected, we may also write that $G$ is disconnected. $V'$ is called a (strongly) connected 
component if $G[V']$ is (strongly) connected and maximal, i.e., if there is no $v\in V\setminus V': V'\cup \{v\}$ 
is (strongly) connected. If $V'=V$ then we say that $G$ is (strongly) connected. 

\paragraph{Cut vertex} For a graph $G\coloneqq (V, E)$ a vertex $v\in V$ is called a cut-vertex if $G[V\setminus \{v\}]$ 
is disconnected. Accordingly, in the directed case, then $v\in V$ is a cut-vertex if $G_\circ[V\setminus \{v\}]$ is 
disconnected. 

\paragraph{Strong blocks} For a digraph $G\coloneqq (V, E)$ a strongly connected subgraph $G'\subseteq G$ is 
called a strongly connected block or strong block if it does not contain a cut-vertex.

\paragraph{Elementary circuit} An elementary circuit of length $k$ in a digraph $G\coloneqq (V,E)$ is a sequence 
of vertices $(v_1,\ldots, v_k, v_1)$ such that $(v_1,\ldots, v_k)$ is a path, $(v_k, v_1) \in E$. 

\paragraph{Digon} A digon is an elementary circuit of length $2$.

\section{RAFs}

\subsection{A Partial Order on RAFs}

The easiest way to define such a partial order is by generating the set
$\closure{R'}{X'}$ iteratively. Let therefore:
\begin{equation}
  c_{\RAF}: 2^X \rightarrow 2^X, c_{\RAF}(X') \coloneqq X' \cup \{\fproducts(\fr) \subseteq X' \ \vert \ \feducts(\fr)\subseteq X'\}. 
\end{equation}
We may drop the subscript $\RAF$ whenever appropriate. In particular, we
write $c(X')$ instead of $c_{\RAF}(X')$ as well as $\closure{}{X'}$ instead
of $\closure{\RAF}{X'}$. \PFS{By construction, $c:2^X \rightarrow 2^X$ is a
  generalized closure operator, i.e., it satisfies, for all $X',X''\in 2^X$
  \begin{itemize}
  \item[(c0)] $c(\emptyset)=\emptyset$,
  \item[(c1)] $X'\subseteq X''$ implies $c(X')\subseteq c(X'')$ (isotone)
  \item[(c2)] $X\subseteq c(X')$ (extensive)
  \end{itemize}
  A set $X\subseteq X'$ is closed (w.r.t.\ $c$) if $X'=c(X')$.  Since $X$
  is finite, iterated application $c$ eventually yields a close set: to see
  this is suffices to note that by (c1) and (c2) we have $X'\subseteq
  c^k(X')\subseteq c^{k+1}(X')$ for all $k\ge 1$. Thus for every $X'\in X$,
  there is $n\in\mathbb{N}$ such that $c^{n+1}(X')=c^{n}(X')$. Moreover,
  comparison of the definitions of $c$ and $\mathscr{C}$ implies that
  $c(\mathscr{C}(X'))=\mathscr{C}(X')$, and conversely $c(X')=X'$ implies
  $X'=\mathscr{C}(X')$. Considering a finite CRS network $(X,\FR,\cat)$
  with $\food\subseteq X, \RAF\subseteq \FR$, there is $n\in\mathbb{N}$
  such that
  \begin{equation}
    \label{eq:seqLimit}
    \closure{\food} = c^k(\food) \qquad\text{for all } k\ge n 
  \end{equation}
}

\begin{tiny} 
It has been shown earlier that the repetitive iteration of $c$ converges to the closure 
in a finite number of steps for finite-sized species sets \cite{Stadler2018, bridging_golnik_2026}. 
However, to keep this contribution self-contained, we include the proof for this statement here. 
We consider only finite-sized CRS, i.e., $\vert X \vert<\infty$ and $\vert \FR \vert<\infty$. By definition, 
the composition $c\circ c$ is well-defined. Moreover, we let $c^n\coloneqq \left(\bigcirc^n_{i=0} \ c \right)$, 
i.e., $c^0(X')\coloneqq X'$.

\vspace{6pt}
\begin{lemma}\label{lem:finSize}
	Let $(X,\FR, \cat)$ be a CRS. For a set $\food \subseteq X$ and $\RAF\subseteq \FR$ there 
	is an $n\in \mathbb{N}$ such that: 
	\begin{equation}
		c^n(\food) = c^m (\food), m\geq n, m\in \mathbb{N}
	\end{equation}
\end{lemma}
\begin{proof}
	Assume this is not the case. Then in each $k$-th step $|c^k(\food)| < |c^{k+1}(\food)|$,
	which implies that $|c^k(\food)| \xrightarrow{k\rightarrow\infty}\infty$. This is a 
	contradiction, given that $|X|<\infty$.
\end{proof}
\vspace{6pt}
It remains to show that $\lim_{k\rightarrow \infty} \cl^k(\food) = \closure{}{\food}$. By definition, for 
any $k\in\mathbb{N}: \cl^k(\food)\subseteq \closure{}{\food}$. In addition, the convergence of $\cl^k(\food)$ 
gurantees that $\lim_{k\rightarrow \infty} \cl^k(\food)\in \mathfrak{C}(\food)$. Thus by minimality 
$\closure{}{\food}\subseteq \lim_{k\rightarrow \infty} \cl^k(\food)$, which yields by mutual inclusion that 
$\lim_{k\rightarrow \infty} \cl^k(\food) = \closure{}{\food}$. With the uniqueness of the limit, the subsequent 
corollary follows.

\vspace{6pt}
\begin{corollary}{cor:seqLimit}
	Let $(X,\FR,\cat)$ be a CRS network of finite size with $\food\subseteq X, \RAF\subseteq \FR$. 
	Then $\closure{}{X'}$ is the unique limit of the sequence
	\begin{equation}
		\left(\left(\bigcirc_{i=1}^k \ c\right)(\food)\right)_{k\in\mathbb{N}}.
	\end{equation}
	In addition, there is an $n\in\mathbb{N}: c^n(\food)=c^m(\food), m\geq n$.
\end{corollary}
\vspace{6pt}
\end{tiny}

The sequence $(c^k(\food))_{k\in\mathbb{N}}$ therefore defines a sequence
of reactions that can be executed in the $k$-th step, i.e., for which all
reactants are contained in the set $\cl^k(\food)$:
\begin{equation}
	d^k(X') \coloneqq \{\fr\in \RAF \ | \ \feducts(\fr)\subseteq c^k(\food)\}
\end{equation}
We can then restrict the sets obtained by $c^k(\food)$ and $d^k(\food)$ to those species and 
reactions that are not present in an ``earlier'' set $c^\ell(\food)$ and $d^\ell(\food), \ell<k$, respectively:
\begin{equation}
  \begin{split}
    X^0(\food')& \coloneqq \food \qquad X^k(\food)\coloneqq c^k(\food)\setminus \left(\bigcup_{i=0}^{k-1} c^k(\food) \right)\\
    \FR^0(\food) & \coloneqq \emptyset \qquad \FR^k(\food)\coloneqq d^k(\food) \setminus \left(\bigcup_{i=0}^{k-1} d^k(\food) \right)
  \end{split}
\end{equation}
These sequences naturally generate a partial order $\preceq$, which we make
explicit by two maps that extract for each species and reaction, the
iteration of their first appearance:
\begin{equation}\label{eq:Gen}
  \begin{split}
    \sgen&: \closure{\RAF}{\food} \rightarrow \mathbb{N}, \sgen(x)\coloneqq k \quad \text{ s. t. } x\in X^k(\PFS{\food}) \\
    \rgen&: \RAF \rightarrow \mathbb{N}, \rgen(r)\coloneqq k \quad \text{ s. t. } r\in \FR^k(\PFS{\food}). 
  \end{split}
\end{equation}
In particular, a species $x_1$ precedes a species $x_2: x_1\preceq x_2$ if and only if $\sgen(x_1) \leq \sgen(x_2)$ 
and a reaction $\fr_1$ precedes a reaction $\fr_2: \fr_1\preceq \fr_2$ if and only if $\rgen(\fr_1) \leq \rgen(\fr_2)$. We refer 
to the values of $\sgen$ and $\rgen$ as the generation of an F-reaction or metabolite. Since for some$\fr\in \FR$ all $r\in \fr$ 
share the same F-reactants, we might also define for $reactions$:
\begin{equation}
  \delta: \varphi^{-1}(\RAF)\rightarrow \mathbb{N}, \delta(r)\coloneqq
  \rgen(\fr)
\end{equation}

The partial order $\preceq$ allows us to easily observe that each species
vertex for the closure of a $\RAF$ is reachable from at least one food
molecule in $\food$ in the bipartite K{\"o}nig graph, which is restricted
to the $\closure{\RAF}{\food}$, the reactions in $\RAF$, and to those
catalyzations $\cat'$ of reactions in $\RAF\subseteq \FR$. Throughout, we
denote this graph as $\queen'\coloneqq \queen(\closure{\RAF}{\food}, \RAF,
\cat')$.

\subsection{A partial Order on RAFs}

\vspace{6pt}
\begin{mainitem}{Lemma}{M-lem:RAFPath}
  Let $(X, \FR, \cat)$ be a CRS with $\food \subseteq X$ and a RAF
  $\RAF\subseteq \FR$.  For any $x\in \closure{}{\food}\setminus \food$
  there is an $f\in \food$ such that $P=(f,\ldots,x)$ is a path in
  $\queen'$.
\end{mainitem}
\begin{proof}
  Let $x\in \closure{}{\food}\setminus \food$. By
  Corollary~\ref{cor:seqLimit} we have for some $n\in\mathbb{N}$ that
  $\closure{}{\food} = c^n(\food)$. Thus, there is the sequence
  $(c^{k-1}(\food), d^k(\food))_{k\in\mathbb{N}}$ such that for any $r\in
  d^k(\food)$ there is $y\in c^{k-1}_{}(\food), z\in c^{k}_{}(\food):
  (y,r)\in E_1(\queen'), (r,z)\in E_2(\queen')$.  Thus, starting with $x\in
  c^n_(\food)$ and recursively applying thisargument, there is a
  sequence of vertices $ (x_0, r_1, x_1, \ldots, r_n, x_n):
  (x_{i-1},r_i)\in E_1(\queen'), (r_i,x_i)\in E_2(\queen'), 1\leq i\leq n,
  x=x_n$ that is a path in $\queen'$.
\end{proof}
\vspace{6pt}

\subsection{Minimal RAFs}

\subsubsection{Strong-connectivity}

\vspace{6pt}
\begin{lemma}\label{lem:RGQSC}
  Let $\RAF$ be an irreducible, monocatalyzed RAF and associated sub-CRS 
  $\mathbf{\Phi'}$ (Eq.~\ref{eq:RAF-CRS}).Then $\RGraph(\mathbf{\Phi}')$ is
  strongly connected.
\end{lemma}
\begin{proof}
Indirectly assume that this is not the case. Then there are at least two non-empty, 
strongly connected components $\RAF_1$ and $\RAF_2$. Assume w.l.o.g. that 
there is no path from any $r_2\in \RAF_2$ to any $r_1\in \RAF_1$. Then all F-reactants 
and F-catalysts of all F-reactions in $\RAF_1$ are in the closure $\closure{\RAF_1}{\food}$, 
in symbols $\feducts(\RAF_1)\cup \cat(\RAF_1)\subseteq \closure{\RAF_1}{\food}$. With 
$\RAF_1\subseteq \RAF$ and $\RAF$ being a RAF, this implies that $\RAF_1$ 
is itself a RAF, which contradicts the minimality of $\RAF$.
\end{proof}
\vspace{6pt}

\begin{mainitem}{Lemma}{M-lem:KSC}
  Let $\RAF$ be a non-trivial, minimal RAF. Then $\RGraph(\king')$ is
  strongly connected.
\end{mainitem}
\begin{proof}
  Lemma~\ref{M-lem:RGQSC} has shown that $\RGraph(\queen')$ is strongly
  connected for a non-trivial minimal RAF. Assume now that
  $\RGraph(\king')$ is not strongly connected, which implies the existence
  of two strongly connected components of $R'\coloneqq \varphi^{-1}(\RAF),
  R'_1$, $R'_2$. Then either $R'_1$ and $R'_2$ partition $R$ such that all
  representatives of an equivalence class are contained in the same
  component, or not. The first case, $\varphi^{-1}(R'_1)\cap
  \varphi^{-1}(R'_2)=\emptyset$, implies that $\RGraph(\queen')$ is not
  strongly connected, which contradicts that $\RAF$ is a non-trivial,
  minimal RAF.  Thus, $\varphi^{-1}(R'_1)\cap \varphi^{-1}(R'_2)=\emptyset$
  and there are $r'\in R'_1$ and $r''\in R'_2$. W.l.o.g., we assume that
  there is no path from any $r_2\in R'_2$ to any $r_1\in R'_1$. Then,
  however, there is also no edge $(r', r_1)$ for any $r_1\in R'_1$ since
  there would otherwise also be an edge $(r'', r_1)$. Hence, $r'\notin
  R'_1$, a contradiction.
\end{proof}

\subsubsection{Cut vertices}

\vspace{6pt}
\begin{mainitem}{Lemma}{M-lem:CutVertexReac}
  Let $\RAF$ be a non-trivial, minimal RAF. If $\princess(\RAF)$ is not a
  strong block, then every cut-vertex is an F-reaction. Accordingly, if
  $\lord(\RAF)$ is not a strong block, then every cut-vertex is a reaction.
\end{mainitem}
\begin{proof}
  Assume there is a cut-vertex $w\in X(\princess)$. Then
  $\princess[V(\princess)\setminus \{w\}]$ is composed of at least two
  connected components $\princess_1$ and $\princess_2$. Then for any two
  vertices $v\in \princess_1$ and $u\in \princess_2$ we have $w\in P_{uv}$
  and $w\in P_{vu}$ for arbitrary paths connecting $u$ and $v$. Therefore,
  there is no $x_1\in X(\princess_1)$ such that $x_1$ is an F-reactant,
  F-product, or F-catalyst for any reaction $r_2\in \FR(\princess_2)$, and
  there is no $x_2\in X(\princess_2)$ such that $x_2$ is a F-reactant,
  F-product, or F-catalyst for any reaction $r_1\in \FR(\princess_1)$.
  Then $\closure{\FR(\princess_1)}{\food}\neq \emptyset$, implying that
  $\FR(\princess_1)$ is a RAF or (not mutually exclusive)
  $\closure{\FR(\princess_2)}{\food}\neq \emptyset$, implying that
  $\FR(\princess_2)$ is a RAF. In either case, this contradicts the
  assumption that $\RAF$ is a non-trivial minimal RAF. For $\lord(\RAF)$
  the proof follows the same argument with the difference that we have
  $\closure{\varphi^{-1}(R(\lord))}{\food}\neq \emptyset$ implies that
  $\varphi^{-1}(\lord)$ is a RAF or (not mutually exclusive)
  $\closure{\varphi^{-1}(R(\lord_2))}{\food}\neq \emptyset$, implying that
  $\varphi^{-1}(\lord)$ is a RAF, and thereby $\RAF$ cannot have been
  minimal.
\end{proof}	
\vspace{6pt}

\begin{mainitem}{Lemma}{M-lem:CutVertPP}
  Let $\RAF$ be a non-trivial, minimal RAF. If $r \in \varphi^{-1}(\RAF)$
  is a cut-vertex in $\lord(\RAF)$ then $|\fr|=1$ and $\fr$ is a cut-vertex
  in $\princess(\RAF)$.
\end{mainitem}
\begin{proof}
  Assume $|\fr|>1$, then there is $r'\in \fr, r\neq r'$. Since $r$ is a
  cut-vertex, it separates $\lord(\RAF)$ in two connected components
  $\lord_1$ and $\lord_2$, both of which contain $r$ and their induced
  subgraphs are strongly connected. Then any path from a vertex in $u\in
  \lord_1$ to a vertex $v\in \lord_2$ and vice versa passes through $r$. We
  then have two cases. First, assume w.l.o.g.\ that there is an F-reactant
  $x \in \lord_1$, and there is an F-product $y\in \lord_2$. Then
  $(x,r',y)$ is a path connecting the two connected components $\lord_1$
  and $\lord_2$, which contradicts that $r$ is a cut-vertex. Assume
  therefore that all F-reactants and F-products are located in one
  connected component, say $\lord_1$, which implies that one F-catalyst $c$
  of $r$ that is neither an F-reactant nor an F-product is located in
  $\lord_2$. Clearly, then $r'\in \lord_1$, i.e., all $r''\in \fr: r'' \in
  \lord_1$. Since $c$ is $\food$ generated, it is a net product of some
  reaction $r_2\in \lord_2$, thus also $\fr_2\in
  \varphi^{-1}(R(\lord_2))$. Except for catalysts for $r$, no other species
  of $\lord_2$ are either reactants for or products of reactions in
  $\lord_1$. Thus, all catalysts for reactions in $\lord_2$ except of $r$
  are in $\lord_2$. Thus $\varphi^{-1}(R(\lord_2)) \setminus \{\fr\}$ is a
  RAF which contradicts minimality of $\RAF$.
\end{proof}

\subsubsection{Strong blocks}

The following statement guarantees the existence of F-reactions in a strong
block that have only F-catalysts as reactants inside the block. We will
refer to them as \emph{pure F-reactions}.
\vspace{6pt}
\begin{mainitem}{Lemma}{M-lem:cutVertexPureCat}
  For a non-trivial minimal RAF $\RAF$, any strong block $\block$ of
  $\princess(\RAF)$ contains a pure F-reaction.
\end{mainitem}
\begin{proof}
  We start with strong blocks $\block$ that contain an F-reaction $\fr:
  \rgen(\fr)=1$. The F-reactants of any of these F-reactions are subsets of
  $\food$. Since $\princess[\block]$ is strongly-connected, it has at least
  one F-catalyst that is not an F-reactant as a reactant. Any other strong
  block $\block'$ contains a cut-vertex $\fr_1\in \RAF$. We start with the
  simple case in which $\block'$ contains only one cut-vertex. We observe
  that $\rgen(\fr_1)< \rgen(\fr)$, for all $\fr\in \FR(B')\setminus
  \{\fr_1\}$. Since $\princess[\block']$ is strongly connected, it has at
  least one reactant and one product. Assume a reactant $x\in X(\block')$
  for $\fr^*$ is also an F-reactant for $\fr^*$. This, however, implies
  that $x$ is an F-product of another F-reaction $\fr_2\in
  \FR(\block')\Rightarrow \rgen(\fr_2)< \rgen(\fr_1)$, a contradiction.

  Assume now that $\block'$ has cut-vertices: $\FR^*(\block')\coloneqq
  \{\fr^*_1, \ldots, \fr^*_k\}$. We may then choose $\fr_j^*:
  \rgen(\fr_j^*)\leq \rgen(\fr^*_i), \forall \ \fr^*_i\in
  \FR^*(\block')$. Assume now that $\fr_j^*$ has an F-reactant $x_j\in
  X(\block')$. Then $x_j$ is an F-product of some reaction in $\fr_1$, and
  $\rgen(\fr_1)<\rgen(\fr^*_j)$. Then $\fr_1$ cannot be a cut-vertex since
  then $\rgen(\fr_1)<\rgen(\fr^*_j)\leq \rgen(\fr_1)$, which is a
  contradiction. Thus, $\fr_1$ is not a cut-vertex and thereby has an
  F-reactant $x_1 \in X(\block')$. Then $x_1$ is an F-product of some
  F-reaction $\fr_2\in \FR(\block')$. If $\fr^*_j=\fr_2$ then
  $\rgen(\fr^*_j)< \rgen(\fr_1)<\rgen(\fr^*_j)$, a
  contradiction. Otherwise, $\rgen(\fr_2)<\rgen(\fr_1)<\rgen(\fr^*_j)$.  By
  iterative application of upper argument, we obtain with the finiteness of
  $\block$:
  \begin{equation}
    \rgen(\fr_k)<\rgen(\fr_{k-1}) <\ldots < \rgen(\fr_l)
  \end{equation}
  with $r_k=r_l$, which yields a contradiction and thereby
  $\rgen(\fr^*_j)\leq \rgen(\fr), \forall \fr\in \FR(\block)\setminus
  \{\fr_1\}$ and $\rgen(\fr^*_j)< \rgen(\fr)$ for all $\fr\in
  \FR(\block)\setminus \FR^*(\block')$.
\end{proof}

\begin{mainitem}{Lemma}{M-lem:MetzCircuit}
  Let $B$ be a strong block of $\lord(\RAF)$. Then $B$ contains an
  elementary circuit $C$ with one \emph{pure reaction} vertex, and there is
  a reordering of the columns of $ \SM[X(C),R(C)]$ that yields a Metzler
  matrix.
\end{mainitem}
\begin{proof}
  We start with the simple case, and assume there is only one \emph{pure
  reaction} $r_1$ in $B$.  By the proof of Lemma~\ref{lem:cutVertexPureCat}
  we have $\delta(r_1)<\delta(r)$ for all $r\in R(B) \setminus \{r_1\}$. We
  may then construct the following elementary circuit:
  \begin{equation}
    r_1 \leftarrow x_1 \leftarrow r_2 \leftarrow x_2 \leftarrow \ldots r_n
    \leftarrow x_m \leftarrow r_{m+1}
  \end{equation}
  where $x_1$ is an F-catalyst of $r_1$, and forall $1< k\leq m+1$: 
  \begin{equation}
    r_k =  \arg \min \{ \delta(r) \ | r\in R(B), \hat{s}_{x_{k-1}r}>0\}
  \end{equation}
  and for all $1<k\leq m$:
  \begin{equation}
    x_k = \arg \min \{\sgen(x) \ \vert \ x\in X(B), s^-_{xr_k}>0\}.
  \end{equation}
  Since all reaction vertices $r_k, 1< k\leq m$ are not \emph{pure
  reaction} vertices, each of them has an F-reactant $x_k$ for which
  $\sgen(x_k)<\delta(r_k)$ and thereby:
  \begin{equation}
    \delta(r_k) < \sgen(x_k) \leq \delta(r_{k+1}), \quad 1\leq k < m.
  \end{equation}
  This generates a strictly monotone decreasing sequence
  $(\rgen(r_k))_{1\leq k\leq m+1}$, which converge due to the finiteness of
  $B$ and we actually obtain that $r_{m+1}=r_1$.  Thus $C\coloneqq
  (r_{m+1}, x_m, r_m, x_{m-1}, r_{m-1},\ldots, x_1,r_1)$ is an elementary
  circuit.
	
  Assume now that there is $i\neq j$, thus $s^-_{x_ir_j}>0$. If $i>j$ then
  $\sgen(x_i)<\sgen(x_j)$ which contradicts that $\sgen(x_j)=\min
  \{\sgen(x) \ \vert \ x\in X(B), s^-_{xr_k}>0\}$. Thus $i<j$, which
  implies that $\sgen(x_i)\geq\delta(r_j)$ and that $x_i$ cannot be an
  F-reactant of $r_j$ but only an F-catalyst, thus:
  \begin{equation}
    \hat{s}_{x_ir_j}=\SM_{x_ir_j}=0.
  \end{equation} 
  Therefore, the only positive educt stoichiometries can be of the form
  $s^-_{x_i,r_i}$.
	
  We can then obtain $S(C)$ by reordering the columns of the submatrix
  $\SM[X(C), R(C)]$ such that $S(C)_{x_iy}\coloneqq \SM[X(C),
    R(C)]_{x_ir_i}$. Thereby, all diagonal entries of $\SM(C)$ are
  nonpositive, i.e., all are negative except $\SM(C)_{x_1r_1}$, and all
  off-diagonal entries of $\SM(C)$ are nonnegative, thus $\SM(C)$ is a
  Metzler matrix.
  \vspace{6pt}
	
  We now turn to the general case and assume there are multiple pure
  reaction vertices $R^*(B)=\{r \in R(B) \ \vert \ \forall x \in X(B):
  s^-_{xr}>0\Rightarrow \hat{s}_{xr}=0\}$.  We choose $r_1\in R^*(B):
  \delta(r_1)\leq \delta(r_j), \forall r_j\in R^*(B)$ and we have
  $\delta(r_1)<\delta(r), \forall r\in R(B)\setminus R^*(B)$. We now apply
  the same construction as above:
  \begin{equation}
    r_1 \leftarrow x_1 \leftarrow r_2 \leftarrow x_2 \leftarrow \ldots r_n \leftarrow x_m \leftarrow r_{m+1}
  \end{equation}
  such that $x_1$ is an F-catalyst of $r_1$, and forall $1< k\leq m+1$:
  \begin{equation}
    r_k = \arg \min \{ \delta(r) \ | r\in R(B), \hat{s}^+_{x_{k-1}>0}\}
  \end{equation}
  and for all $1<k\leq m$:
  \begin{equation}
    x_k = \arg \min \{\delta (x) \ \vert \ x\in X(B), s^-_{xr_k}>0\}
  \end{equation}
  if $r_k \notin R^*(B)$ or $x_k$ as an F-catalyst of $r_k$ otherwise. We
  thereby obtain the following structure:
  \begin{equation*}
    (r_{l_0}, \leftarrow x_{l_0} \leftarrow\ldots \leftarrow x_{l_1})
    \leftarrow \ldots \leftarrow (r_{l_p} \leftarrow x_{l_p} \leftarrow
    \ldots \leftarrow x_{l_{p+1}}) \leftarrow r_{l_{p+1}}
  \end{equation*}
  where $r_{l_0} \coloneqq r_1, x_{l_0} \coloneqq x_1$ and each reaction
  vertex $r_{l_i+1}, 0\leq i\leq p$ is a pure reaction, and each species
  vertex $x_{l_i+1}, 0\leq i\leq p$ is an F-catalyst for the pure reaction
  $r_{l_i+1}$. Then, by the same argument as above, we obtain a strictly
  monotone decreasing sequence $(\delta(r_j))_{l_i\leq j \leq
    _{l_{i+1}-1}}$ for each $(r_{l_i} \leftarrow x_{l_1+1} \leftarrow
  \ldots x_{l_{i+1}}), 0\leq i\leq p$. By finiteness of $B$, we then obtain
  that $r_{l_i}=r_{l_{p+1}}, 0\leq i\leq p$ yields an elementary circuit
  \begin{equation}
    C\coloneqq (r_{l_{p+1}}, x_{l_{p+1}}, \ldots x_{l_i},  r_{l_i})
  \end{equation}
  \vspace{6pt}
  
  With the same argument as above, for each $i\neq j: s^-_{x_ir_j}>0$ such
  that there is a $k\in \{0,\ldots, p\}: l_k\leq i,j \leq l_{k+1}$ we
  obtain that $\hat{s}_{x_ir_j}=0$.
	
  Assume now that there is no such $k$. If $\sgen(x_i)>\sgen(x_j)$ then
  $\sgen(x_i)\geq \delta(r_j)$ which implies that $x_i$ cannot be an
  F-reactant for $r_j$, thus $\hat{s}_{x_ir_j}=0$. The case
  $\sgen(x_i)<\sgen(x_j)$ is impossible due to our choice of $x_j$. Then
  the only remaining case is $\sgen(x_i)=\sgen(x_j)$ which is possible
  since the minimum of $\{\sgen(x) \ \vert \ x\in X(B), s^-_{xr_j}>0\}$
  might not be unique. This, however, implies that there is a smaller
  elementary circuit containing a pure reaction vertex. If $i>j$ this
  circuit can be obtained as $C'$ with:
  \begin{equation}
    \begin{split}
      V(C')&\coloneqq V(C)\setminus \{x_i, r_i, \ldots, x_j\}\\[0.5em]
      E(C')& \coloneqq E(C)\setminus \{(x_k, r_k) \ | \ j-1\leq k \leq i\} \cup \{(r_k, x_{k+1}) \ | \ j\leq k \leq i\}
    \end{split}
  \end{equation} 
  or if $i<j$ then this circuit can be obtained as $C''$:
  \begin{equation}
    \begin{split}
      V(C'') & \coloneqq \{r_j, x_{j-1}, \ldots, x_i\} \\
      E(C'') & \coloneqq\{(x_k, r_k) \ | \ i \leq k \leq j-1\} \cup \{(x_i,r_j)\} \cup \{(r_k, x_{k-1}) \ | \ i+1\leq k \leq j\}
    \end{split}
  \end{equation} 
  It holds by construction that either $r_1\in R(C')$ or that there is a
  pure reaction vertex $r_{l_k} \in R(C''): i<l_k<j$. We may now repeat
  this procedure until we can not find such a metabolite-to-reaction edge
  anymore and obtain a final elementary circuit $\tilde{C}$.
	
  We can then obtain $S(\tilde{C})$ by reordering the columns of the
  submatrix $\SM[X(\tilde{C}), R(\tilde{C})]$ such that
  \begin{equation}
    \SM(\tilde{C})_{xy}\coloneqq \SM[X(\tilde{C}), R(\tilde{C})]_{xv^{\tilde{C}}_{out}(x)}
  \end{equation} 
  where $v^{\tilde{C}}_{out}(x)$ is the unique reaction $r\in R(\tilde{C}$
  such that $(x,r)\in E(\tilde{C})$. Thereby, all diagonal entries of
  $\SM(\tilde{C})$ are nonpositive, i.e., all are negative except
  $\SM(\tilde{C})_{x_1r_1}$, and all off-diagonal entries of
  $\SM(\tilde{C})$ are nonnegative, thus $\SM(\tilde{C})$ is a Metzler
  matrix.
\end{proof}

\begin{mainitem}{Lemma}{M-lem:MetzCircAut}
	If $C$ is a Metzler circuit containing a pure reaction, then $(X(C), R(C)$ 
	is stoichiometrically autocatalytic.
\end{mainitem}
\begin{proof}
	If $C$ is a Metzler circuit that contains a pure reaction, then there is a reordering of the columns of 
	the stoichiometric submatrix $\SM[X(C),R(C)]$ that reads
	\begin{equation}
			\SM(C) = 
				\begin{pmatrix}
						0 	& 	 		&	 			&  			& 	p_1	\\
						p_2	& 	-r_2  		& 				&     P''  		&		\\
					 		& 	p_3  		& 	\ddots 		&      			&		\\
							& 	 		& 	\ddots 		& \ddots 		&    	 	\\
						P'	& 	 		& 				& p_k 		& -r_k     	\\
				\end{pmatrix},
	\end{equation}

i.e., $\SM(C)$ with $\{r_i\}\in \mathbb{R}_{\geq 0}, i=2,\ldots, k$ and $\{p_i\}\in \mathbb{R}_{>0},i=1,\ldots, k$ 
and non-negative entries in $P'$ and $P''$. We may then construct $v\in \mathbb{R}^{k}_{>0}$ as $v(r_k)\coloneqq 1$ 
and 
\begin{equation}
	v_{i}=\begin{cases}
			\frac{v_{i+1}\cdot r_{i+1}}{p_{i+1}} +1 & \text{if } r_{i+1}>0 \\
			\frac{v_{i+1}}{p_{i+1}} +1 & \text{if } r_{i+1}=0
		\end{cases}
\end{equation}
Then $v>0$ and by construction we directly obtain $\SM(C)v>0$ since $P'$
and $P''$ contain only non-negative entries \PFS{and very row of $S(C)$
  contains a strictly positive entry}.
\end{proof}
