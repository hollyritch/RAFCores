\section{RAFs}

\subsection{A Partial Order on RAFs}

The easiest way to define such a partial order is by generating the set $\closure{R'}{X'}$ iteratively. Let therefore:
\begin{equation}
	c_{\RAF}: 2^X \rightarrow 2^X, c_{\RAF}(X') \coloneqq X' \cup \{\fproducts(\fr) \subseteq X' \ \vert \ \feducts(\fr)\subseteq X'\}. 
\end{equation}
We may drop the subscript $\RAF$ whenever appropriate. In particular, we write $c(X')$ instead 
of $c_{\RAF}(X')$ as well as $\closure{}{X'}$ instead of $\closure{\RAF}{X'}$.

It has been shown earlier that the repetitive iteration of $c$ converges to the closure 
in a finite number of steps for finite-sized species sets \cite{Stadler2018, bridging_golnik_2026}. 
However, to keep this contribution self-contained, we include the proof for this statement here. 
We consider only finite-sized CRS, i.e., $\vert X \vert<\infty$ and $\vert \FR \vert<\infty$. By definition, 
the composition $c\circ c$ is well-defined. Moreover, we let $c^n\coloneqq \left(\bigcirc^n_{i=0} \ c \right)$, 
i.e., $c^0(X')\coloneqq X'$.

\vspace{6pt}
\begin{lemma}\label{lem:finSize}
	Let $(X,\FR, \cat)$ be a CRS. For a set $X'\subseteq X$ and $\RAF\subseteq \RAF$ there 
	is an $n\in \mathbb{N}$ such that: 
	\begin{equation}
		c^n(X') = c^m (X'), m\geq n, m\in \mathbb{N}
	\end{equation}
\end{lemma}
\begin{proof}
	Assume this is not the case. Then in each $k$-th step $|c^k(X')| < |c^{k+1}(X')|$,
	which implies that $|c^k(X')| \xrightarrow{k\rightarrow\infty}\infty$. This is a 
	contradiction, given that $|X|<\infty$.
\end{proof}
\vspace{6pt}
It remains to show that $\lim_{k\rightarrow \infty} \cl^k(X') = \closure{}{X'}$. By definition, for 
any $k\in\mathbb{N}: \cl^k(X')\subseteq \closure{}{X'}$. In addition, the convergence of $\cl^k(X')$ gurantees 
that $\lim_{k\rightarrow \infty} \cl^k(X')\in \mathfrak{C}(X')$. Thus by minimality $\closure{}{X'}\subseteq 
\lim_{k\rightarrow \infty} \cl^k(X')$, which yields by mutual inclusion that $\lim_{k\rightarrow \infty} \cl^k(X') 
= \closure{}{X'}$. With the uniqueness of the limit, the subsequent corollary follows.

\vspace{6pt}
\begin{corollary}\label{cor:seqLimit}
	Let $(X,\FR,\cat)$ be a CRS network of finite size with $X'\subseteq X, \RAF\subseteq \FR$. 
	Then $\closure{}{X'}$ is the unique limit of the sequence
	\begin{equation}
		\left(\left(\bigcirc_{i=1}^k \ c\right)(X')\right)_{k\in\mathbb{N}}.
	\end{equation}
	In addition, there is an $n\in\mathbb{N}: c^n(X')=c^m(X'), m\geq n$.
\end{corollary}
\vspace{6pt}
The sequence $(c^k(X')_{k\in\mathbb{N}}$ directly yields a sequence of reactions that can be
executed in the $k$-th step, i.e., for which all reactants are contained in the set $\cl^k(X')$: 
\begin{equation}
	d^k(X') \coloneqq \{r\in \RAF \ | \ \feducts(r)\subseteq c^k(X')\}
\end{equation}
We can then restrict the sets obtained by $c^k(X')$ and $d^k(X')$ to those species and 
reactions that are not present in an ``earlier'' set $c^\ell(X')$ and $d^\ell(X'), \ell<k$, respectively:
\begin{equation}
	\begin{split}
		\POX^0(X')& \coloneqq X' \qquad \POX^k(X')\coloneqq c^k(X)\setminus \left(\bigcup_{i=0}^{k-1} c^k(X) \right)\\
		\POR^0(X') & \coloneqq \emptyset \qquad \POR^k(X')\coloneqq d^k(X) \setminus \left(\bigcup_{i=0}^{k-1} d^k(X) \right)
	\end{split}
\end{equation}
These sequences naturally generate a partial order $\preceq$, which we make explicit by two maps 
that extract for each species and reaction the iteration of their first appearance: 
\begin{equation}\label{eq:Gen}
	\begin{split}
		g&: X' \rightarrow \mathbb{N}, g(x)\coloneqq k \quad \text{ s. t. } x\in \POX_{R'}^k(X') \\
		h&: R' \rightarrow \mathbb{N}, h(r)\coloneqq k \quad \text{ s. t. } r\in \POR_{R'}^k(X'). 
	\end{split}
\end{equation}
In particular, a species $x_1$ precedes a species $x_2: x_1\preceq x_2$ if and only if $g(x_1) \leq g(x_2)$ 
and a reaction $\fr_1$ precedes a reaction $\fr_2: \fr_1\preceq \fr_2$ if and only if $h(\fr_1) \leq h(\fr_2)$. We refer 
to the values of $g$ and $h$ as the generation of a reaction or metabolite. 

The partial order $\preceq$ allows us to easily observe that each species vertex for the closure of a 
$\RAF$ is reachable from at least one food molecule in $\food$ in the bipartite K\"onig graph, 
which is restricted to the $\closure{\RAF}{\food}$, the reactions in $\RAF$, and to those catalyzations 
$\cat'$ of reactions in $\RAF\subseteq \FR$. Throughout, we denote this graph as 
$\queen'\coloneqq \queen(\closure{\RAF}{\food}, \RAF, \cat')$.

\subsection{A partial Order on RAFs}

\vspace{6pt}
\begin{lemma}\label{M-lem:RAFPath}
	Let $(X, \FR, \cat)$ be a CRS with $\food \subseteq X$ and a RAF $\RAF\subseteq \FR$. 
	For any $x\in \closure{}{\food}\setminus \food$ there is an $f\in \food$ such that $P=(f, \ldots, x)$ 
	is a path in $\queen'$.
\end{lemma}
\begin{proof}
	Let $x\in \closure{}{\food}\setminus \food$. By Corollary~\ref{cor:seqLimit} we have for some $n\in\mathbb{N}$ that
	$\closure{}{\food} = c^n(\food)$. Thus, there is the sequence $(c^{k-1}(\food), d^k(\food))_{k\in\mathbb{N}}$
	such that for any $r\in d^k(\food)$ there is $y\in c^{k-1}_{}(\food), z\in c^{k}_{}(\food): (y,r)\in E_1(\queen'), (r,z)\in E_2(\queen')$. 
	Thus, starting with $x\in c^n_(\food)$ and recursively applying the upper argument, there is a sequence of vertices $
	(x_0, r_1, x_1, \ldots, r_n, x_n): (x_{i-1},r_i)\in E_1(\queen'), (r_i,x_i)\in E_2(\queen'), 1\leq i\leq n, x=x_n$ that is a path 
	in $\queen'$.
\end{proof}
\vspace{6pt}

\subsection{Minimal RAFs}

\subsubsection{Strong-connectivity}

\begin{lemma}\label{lem:KSC}
	Let $\RAF$ be a non-trivial, minimal RAF. Then $\RGraph(\king')$ is strongly connected.
\end{lemma}
\begin{proof}
	Lemma~\ref{lem:RGQSC} has shown that $\RGraph(\queen')$ is strongly connected for a non-trivial 
	minimal RAF. Assume now that $\RGraph(\king')$ is not strongly connected, which implies the existence 
	of two strongly connected components of $R'\coloneqq \varphi^{-1}(\RAF), R'_1, R'_2$. Then either $R'_1$ 
	and $R'_2$ partition $R$ such that all representatives of an equivalence class are contained in the same 
	component, or not. The first case, $\varphi^{-1}(R'_1)\cap \varphi^{-1}(R'_2)=\emptyset$, implies that 
	$\RGraph(\queen')$ is not strongly connected, which contradicts that $\RAF$ is a non-trivial, minimal RAF. 
	Thus, $\varphi^{-1}(R'_1)\cap \varphi^{-1}(R'_2)=\emptyset$ and there are $r'\in R'_1, r''\in R'_2$. W.l.o.g.
	we assume that there is no path from any $r_2\in R'_2$ to any $r_1\in R'_1$. Then, however, there is also
	no edge $(r', r_1)$ for any $r_1\in R'_1$ since there would otherwise also be an edge $(r'', r_1)$. Hence,
	$r'\notin R'_1$, a contradiction.
\end{proof}

\subsubsection{Cut vertices}

\vspace{6pt}
\begin{lemma}\label{lem:CutVertexReac}
	Let $\RAF$ be a non-trivial, minimal RAF. If $\princess(\RAF)$ is not a strong block, then every 
	cut-vertex is an F-reaction. Accordingly, if $\lord(\RAF)$ is not a strong block, then every 
	cut-vertex is a reaction.
\end{lemma}
\begin{proof}
	Assume there is a cut-vertex $w\in X(\princess)$. Then $\princess[V(\princess)\setminus \{w\}]$ is composed of at least two connected
	components $\princess_1,\princess_2$. Then for any two vertices $v\in \princess_1$ and $u\in \princess_2$ we have 
	$w\in P_{uv}$ and $w\in P_{vu}$ for arbitrary paths connecting $u$ and $v$. Therefore, there is no $x_1\in X(\princess_1)$ 
	such that $x_1$ is an F-reactant, F-product, or F-catalyst for any reaction $r_2\in \FR(\princess_2)$, and there is no 
	$x_2\in X(\princess_2)$ such that $x_2$ is a F-reactant, F-product, or F-catalyst for any reaction $r_1\in \FR(\princess_1)$. 
	Then $\closure{\FR(\princess_1)}{\food}\neq \emptyset$, implying that $\FR(\princess_1)$ is a RAF or (not mutually exclusive) 
	$\closure{\FR(\princess_2)}{\food}\neq \emptyset$, implying that $\FR(\princess_2)$ is a RAF, in any case, contradicting that  
	$\RAF$ is a non-trivial, minimal RAF. For $\lord(\RAF)$ the proof follows the same argument with the difference that we have 
	$\closure{\varphi^{-1}(R(\lord))}{\food}\neq \emptyset$ implies that $\varphi^{-1}(\lord)$ is a RAF or (not mutually exclusive) 
	$\closure{\varphi^{-1}(R(\lord_2))}{\food}\neq \emptyset$, implying that $\varphi^{-1}(\lord)$ is a RAF, and thereby $\RAF$ 
	was not minimal.
\end{proof}	
\vspace{6pt}

\begin{lemma}\label{lem:CutVertPP}
	Let $\RAF$ be a non-trivial, minimal RAF. If $r \in \varphi^{-1}(\RAF)$ is a cut-vertex
	in $\lord(\RAF)$ then $|\fr|=1$ and $\fr$ is a cut-vertex in $\princess(\RAF)$.
\end{lemma}
\begin{proof}
	Assume $|\fr|>1$, then there is $r'\in \fr, r\neq r'$. Since $r$ is a cut-vertex, it separates 
	$\lord(\RAF)$ in two connected components $\lord_1, \lord_2$, which both contain
	$r$ and their induced subgraphs are strongly connected. Then any path from a vertex in 
	$u\in \lord_1$ to a vertex $v\in \lord_2$ and vice versa passes through $r$. We then
	have two cases. First, assume w.l.o.g. that there is an F-reactant $x \in \lord_1$, and there is 
	an F-product $y\in \lord_2$. Then $(x,r',y)$ is a path connecting the two connected 
	components $\lord_1$ and $\lord_2$, which contradicts that $r$ is a cut-vertex. Assume
	therefore that all F-reactants and F-products are located in one connected component, e.g.
	$\lord_1$, which implies that one F-catalyst $c$ of $r$ that is neither an F-reactant nor an 
	F-product is located in $\lord_2$. Clearly, then $r'\in \lord_1$, i.e., all $r''\in \fr: r'' \in 
	\lord_1$. Since $c$ is $\food$ generated, it is a net product of some reaction $r_2\in \lord_2$, 
	thus also $\fr_2\in \varphi^{-1}(R(\lord_2))$. Except for catalysts for $r$, no other species 
	of $\lord_2$ are either reactants for or products of reactions in $\lord_1$. Thus, all catalysts
	for reactions in $\lord_2$ except of $r$ are in $\lord_2$. Thus $\varphi^{-1}(R(\lord_2))
	\setminus \{\fr\}$ is a RAF which contradicts minimality of $\RAF$.
\end{proof}

\subsubsection{Strong blocks}

We next show that cut-vertices in $\princess(\RAF)$ that have only F-catalysts as reactants play an
 important role in the partial order of a strong block. To this end, we first define the set of 
 \emph{pure reactions} as 
\begin{equation}\label{eq:pureReactions}
	\FR^*(\block)\coloneqq \{\fr\in \FR(\block) \ | \ s^-_{x\fr}>0 \Rightarrow s^+_{x\fr}=s^-_{x\fr}, \forall x\in X(\block)\}.
\end{equation}
In particular, there is always at least one \emph{pure reaction} that has the lowest generation number of all 
reactions within the block, if $R^*(\block)$ is not empty.
\vspace{6pt}
\begin{lemma}\label{lem:genPureReacs}
	Let $\block$ be a strong block of $\princess(\RAF)$. If $\FR^*(\block)\neq \emptyset$, then there is $\fr^*\in R(B*): 
	\rgen(\fr^*)\leq \rgen(\fr), \forall~\fr\in \FR(\block)$ and $\rgen(\fr^*)<\rgen(\fr), \forall \fr\in \FR(\block)\setminus \FR^*(\block)$.
\end{lemma}
\begin{proof}
	Let $\block$ be a strong block of $\princess(\RAF)$. We choose an $\fr_1\in \FR^*(\block)$ and assume 
	$|\FR^*(\block)|=1$. Assume now there is an $\fr\in \FR(\block)\setminus \{\fr_1\} : \rgen(\fr_2) < \rgen(\fr_1)$. 
	Then $\fr_2$ has an F-reactant $x_2 \in X(\block)$ since $\block$ is strongly connected and $\fr_2\notin \FR^*(\block)$. 
	Thus, there is a reaction $\fr_3$ for which $x$ is an F-product. If $\fr_1=\fr_3$ then $\rgen(\fr_1)< \rgen(\fr_2)<\rgen(\fr_1)$, 
	a contradiction. Otherwise, we simply have that $\rgen(\fr_3)<\rgen(\fr_2)<\rgen(\fr_1)$. Then $\fr_3$ has an F-reactant 
	$x_3\in X(\block)$, since $\fr_3\notin \FR^*(\block)$. Then the same argument holds for $\fr_3$ and, by iteratively applying 
	this argument, we obtain with the finiteness of $\block$:
	\begin{equation}
		\rgen(r_k)<\rgen(r_{k-1} <\ldots < \rgen(r_l)
	\end{equation}
	with $r_k=r_l$, which yields a contradiction and thereby $\rgen(\fr_1)<\rgen(\fr), \forall \fr\in \FR(\block)\setminus \{\fr_1\}$. 
	Assume now $|\FR^*(\block)|>1$. We may then chose $\fr_1\in R^*(B): \rgen(\fr_1)\leq \rgen(\fr'), \forall \fr'\in \FR^*(\block)$. 
	We can apply the same iterative argument as above and obtain the same contradiction, with the slight difference that we can
	only deduce that $\rgen(\fr_1)\leq \rgen(\fr), \forall \fr\in \FR(\block), \rgen(\fr_1)<\rgen(\fr), \forall \fr\in \FR(\block) \setminus \FR^*(\block)$.
\end{proof}
\vspace{6pt}

\vspace{6pt}
\begin{lemma}\label{lem:cutVertexPureCat}
	For a non-trivial, minimal RAF $\RAF$ any strong block $\block$ of $\princess(\RAF)$ contains an F-reaction 
	$\fr\in \FR(\block)$ such that all reactants for $\fr$ that are located in $\block$ are pure F-catalysts, in symbols:
	\begin{equation}
		 \forall x\in X(B): s^-_{x\fr}>0: s^-_{x\fr}=k_{x\fr}
	\end{equation}
\end{lemma}
\begin{proof}
	We start with strong blocks $\block$ which contain a reaction vertex $\fr: \rgen(\fr)=1$. The F-reactants of any of these 
	reactions are subsets of $\food$. Since $\princess[\block]$ is strongly-connected, it has at least one F-catalyst that is 
	not an F-reactant as a reactant. Any other strong block $\block'$ contains a cut-vertex $\fr^*\in \RAF$. We start with 
	the simple case in which $\block'$ contains only one cut-vertex. We observe that $\rgen(r^*)< \rgen(r), \ \forall \ r\in 
	\FR(B')\setminus \{r^*\}$. Since $\princess[\block']$ is strongly connected, it has at least one reactant and one 
	product. Assume a reactant $x\in X(B')$ for $\fr^*$ is als an F-reactant for $\fr^*$. This, however, implies that $x$ is 
	an F-product of a reaction $r\in \FR(\block')\Rightarrow \rgen(r)< \rgen(r^*)$, a contradiction. Assume now that $\block'$ has 
	cut-vertices: $r^*_1, \ldots, r^*_k$. Again, we observe, that there is an $r^*_j: \rgen(r^*_j)< \rgen(r), \forall \ r\in 
	\FR(\block') \setminus \{r^*_j\}$. Assume now that $r_j^*$ has an F-reactant $x_j\in X(\block')$. Then $x_j$ is a product of 
	some reaction in $r_i$, and it follows that $\rgen(r_i)<\rgen(r^*_j)$. This contradicts the previous observation, and
	the claim follows.
\end{proof}
