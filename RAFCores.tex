\section{Notation}

A reaction network $(X,R)$ is tuple of two finite sets, set of metabolites $X$ 
and a set of reactions $R$. For any $r\in R$ we can write a stoichiometric equation as:
	\begin{equation}
		\sum_{x\in X}s^-_{xr} \cdot x \rightarrow s^+_{xr}\cdot x
	\end{equation}
where $s^{\pm}_{xr}$ are the educt and product stoichiometry of the molecule $x$ in 
the reaction $r$. The \emph{stoichiometric matrix} $\SM$ encodes all net-stoichiometries:
\begin{equation}
	\SM_{xr}\coloneqq s^+_{xr}-s^-_{xrs}
\end{equation}
 We denote by 
	\begin{equation}
		\begin{split}
			E: R \rightarrow X, E(r) & \coloneqq \{x\in X \ \vert \ s^-_{xr}-s^+_{xr}>0\} \\
			P: R \rightarrow X, P(r) & \coloneqq \{x\in X \ \vert \ s^-_{xr}-s^+_{xr}<0\} \\
		\end{split}
	\end{equation}
the reactants (educts) and products of reaction $r$, respectively. A catalyst is an entity $x\in X$ such 
that $s^{-}_{xr}\cdot s^{+}_{xr}\neq 0$ and we say that $r$ is explicitly catalyzed 
by $x$. In case $s^-_{xr}>s^+_{xr}>0$, $x$ is both reactant and catalyst 
of reaction $r$, while in case $s^+_{xr}>s^-_{xr}>0$, $x$ is both product and catalyst 
of reaction $r$. Note that a reaction may have multiple catalysts. We collect all catalysts 
for a reaction $r$ in the set $\cat_r$ and denote by $\cat(R)\coloneqq\{(K_r,r) \ | \ r\in R 
\text{ explicitly catalyzed}\}$ the set of \emph{catalyzations}. The catalysts of a \emph{catalyzation} 
are denoted by $X(\cat_r)$. The triple $(X,R,\cat)$ with $\cat=\cat(R)$ is called a catalyzed RN. Note
that there may be uncatalyzed reactions in a catalyzed RN.

A tuple $(X', R')$ is called a subnetwork of $(X,R)$ if $X'\subseteq X, R'\subseteq R$, and  
accordingly, $(X', R',\cat')$ is called a catalyzed subnetwork if additionally $\cat'=\cat(R')$. We 
say a subnetwork $(X',R')$ is well formed if for any $r\in R': \exists x_1,x_2\in X: 
s^-_{x_1r}>0, s^+_{x_2r}>0$. Considering a subnetwork $(X',R')$, we may distinguish 
a certain subset of entities $F', W'\subseteq X$ defined as
\begin{equation}
	\begin{split}
		F'&\coloneqq \{x\in X: x\notin X', \exists r\in R': s^{-}_{xr}>0\}\\
		W'&\coloneqq \{x\in X: x\notin X', \exists r\in R': s^{+}_{xr}>0\}.
	\end{split}
\end{equation}

We call $F'$ and $W'$ the set of food and waste molecules for the subnetwork $(X',R')$,
respectively. The closure of a set $X'$ w.r.t. to a set $R'$ denoted as $\closure{R'}{X'}$ is 
the unique minimal set $Y$ such that for any $r\in R'$: 

\begin{equation}\label{eq:Closure}
	E(r)\subseteq X'\cup Y \Rightarrow P(r)\subseteq Y.
\end{equation}

Note that $\closure{R'}{X'}$ does not necessarily contain the set $X'$ itself. However, it will be convenient 
to obtain the set $\closure{R'}{X'}$ iteratively. We therefore define:
\begin{equation}
	\Phi_{R'}: X \rightarrow X, \Phi_{R'}(Z) \coloneqq Z \cup \{P(r) \subseteq Z \ \vert \ E(r)\subseteq Z\}. 
\end{equation}

It is indeed easy to verify that the repetitive iteration of $\Phi_{R'}(X')$ yields $\closure{R'}{X'}\cup X'$. To this end, recall 
that an RN is composed of two finite-sized sets, i.e., $|X|<\infty$ and $|R|<\infty$. The intuitive assumption that 
a repetitive application of $\Phi_{R'}$ actually converges is indeed true.
\vspace{6pt}
\begin{lemma}\label{lem:finSize}
	Let $(X,R)$ be a reaction network. For a set $X'\subseteq X$ and $R'\subseteq R$ there 
	is an $n\in \mathbb{N}$ such that: 
	\begin{equation}
		\left(\bigcirc_{i=1}^n \Phi_{R'}\right)(X') =\left(\bigcirc_{i=1}^{m} \Phi_{R'}\right)(X'), m\geq n, m\in \mathbb{N}
	\end{equation}
\end{lemma}
\begin{proof}
	Assume this is not the case. Then in each $k$-th step 
	\begin{equation}
		|\left(\bigcirc_{i=1}^k \Phi_{R'}\right)(X')| < |\left(\bigcirc_{i=1}^{k+1} \Phi_{R'}\right)(X')|,
	\end{equation}
	which implies that
	\begin{equation}
		|\left(\bigcirc_{i=1}^k \Phi_{R'}\right)(X')| \overset{k\rightarrow\infty}{\longrightarrow}{\infty}	
	\end{equation}
	a contradiction given that $|X|<\infty$.
\end{proof}

The sequence $(Z_k)_{k\in\mathbb{N}} \coloneqq X'\cup \left(\left(\bigcirc_{i=1}^k \Phi_{R'}\right)(X')\right)_{k\in \mathbb{N}}$ thereby converges. We let
$\mathcal{Z}\coloneqq X'\cup \lim_{k\rightarrow \infty} (Z_k)_{k\in \mathbb{N}}$. By definition of $\closure{R'}{X'}$ for any $k\in \mathbb{N}: Z_k \subseteq \closure{R'}{X'}$. 
Assume now that there is $x\in \closure{R'}{X'} \cup X'$ such that $x\notin \mathcal{Z}$. Then, trivially, $x\notin X'$, and there is an $r\in R': x \in P(r)$ 
for which $E(r), P(r)\subseteq \closure{R'}{X'}$. This implies, however, that for all $y\in E(r)$ it holds that either $y\in X'$ or there is $r_y: y\in P(r_y=$ with $E(r_y), P(r_y)
\subseteq \closure{R'}{X'}$. Repeating this argument recursively until all $y\in X'$, yields that there is a $k\in \mathbb{N}$ such that $x\in \left(\bigcirc_{i=1}^k \Phi_{R'}\right)(X')$, a contradiction. 
We thereby obtain that $\mathcal{Z} = X'\cup \closure{R'}{X'}$. Uniqueness of the limit then yields:
\vspace{6pt}
\begin{corollary}\label{cor:seqLimit}
	Let $(X,R,\cat)$ be a catalyzed network of finite size with $X'\subseteq X, R'\subseteq R$. Then $X'\cup \closure{R'}{X'}$ is the unique limit of the sequence
	\begin{equation}
		\left(\left(\bigcirc_{i=0}^k \Phi_{R'}\right)(X')\right)_{k\in\mathbb{N}}.
	\end{equation}
	In addition, there is an $n\in\mathbb{N}: \left(\bigcirc_{i=0}^n \Phi_{R'}\right)(X')=\left(\bigcirc_{i=0}^m \Phi_{R'}\right)(X'), m\geq n$.
\end{corollary}
\vspace{6pt}
The sequence $(Z_k)_{k\in\mathbb{N}}$ implies a sequence of reactions that can be applied in the $k$-th step, i.e. 
$(\Upsilon_{R'}^k(X'))_{k\in\mathbb{N}}$ with $(\Upsilon_{R'}^k(X')) \coloneqq \{r\in R' \ | \ E(r)\subseteq Z_k\}$.

A (catalyzed) RN can be represented by its \emph{K\"onig graph}, a directed bipartite graph $\king (X,R,\cat) \coloneqq 
(X\cup R, E_1\cup E_2)$ with
\begin{equation}
	\begin{split}
		E_1 & \coloneqq \{(x,r) \ \vert \ s^-_{xr}>0\} \\
		E_2 & \coloneqq \{(r,x) \ \vert \ s^+_{xr}>0\}.
	\end{split}
\end{equation}

To align with the setting of \cite{hordijk_detecting_2004}, we define a slight variation denoted as \emph{catalyzed 
K\"onig graph} as the bipartite directed graph $\queen (X,R,\cat) \coloneqq (X\cup R, E_1\cup E_2\cup E_{\cat})$

\begin{equation}
	\begin{split}
		E_1 & \coloneqq \{(x,r) \ \vert \ \SM_{xr}<0\} \\
		E_2 & \coloneqq \{(r,x) \ \vert \ \SM_{xr}>0\} \\
		E_{\cat} & \coloneqq \{(x,r) \ \vert \ x\in X(\cat_r)\}.
	\end{split}
\end{equation}

First, we give the formal definition of a reflexive autocatalytic food-generated system (RAF) provided in 
\cite{hordijk_detecting_2004}.
\vspace{6pt}
\begin{definition}[RAF]\label{def:RAF}
	Let $(X,R,\cat)$ be a catalyzed network with $F\subseteq X$. Then
	a set $R'$ is called  
		\begin{itemize} 
			\item[i.]{\emph{reflexive autocatalytic} (RA) if $\forall r\in R': \emptyset \neq X(\mathcal{K}_r)\subseteq E(R')\cup P(R')$}
			\item[ii.]{\emph{F-generated} if $E(R')\subseteq \closure{R'}{F}$}
			\item[iii.]{\emph{reflexive autocatalytic} and \emph{F-generated} (RAF) if both i. and ii. hold}
		\end{itemize}
\end{definition}
\vspace{6pt}
Note that $ii.$, together with Eq.\ref{eq:Closure}, directly implies that all products for any reaction $r\in R'$ are also contained 
in the closure of $F'$, thus $P_r\subseteq \closure{R'}{F}$. In addition, the set of catalysts $\cat(R')$ is indeed also a subset 
of $\closure{R'}{F}$. 
\vspace{6pt}
\begin{lemma}\label{lem:catClose}
	Let $(X,R,\cat)$ be a catalyzed network with $F\subseteq X$ and $R'$ an RAF. Then $\cat(R') \subseteq \closure{R'}{F}$.
\end{lemma}
\begin{proof}
	Let $c\in \cat(R')$. Then there is a reaction $r\in R': c\in \cat_r$. If $c\in F \Rightarrow c\in \closure{R'}{F}$. If $c\notin F$.
	then $c$ is F-generated, and thus by Def.~\ref{def:RAF}i. $c\in E(R')\cup P(R')\subseteq \closure{R'}{F}$. 
\end{proof}	
\vspace{6pt} Next, we translate the notion of a closure in a graph-theoretical setting. 
\vspace{6pt}
\begin{lemma}\label{lem:RAFPath}
	Let $(X,R,\cat)$ be a catalyzed network with $F\subseteq X$ and $R'$ an RAF. Then, 
	for any $x\in \closure{R'}{F}\setminus F: \exists f\in F: P=(f, \ldots, x)$ is a path in 
	$\king'\coloneqq \king(\closure{R'}{F}\cup F, R', \cat')$.
\end{lemma}
\begin{proof}
	Let $x\in \closure{R'}{F}\setminus F$. By Corollary~\ref{cor:seqLimit} we have for some $n\in\mathbb{N}$ that
	$\closure{R'}{F}\setminus F = \left(\bigcirc_{i=0}^n \Phi_{R'}\right)(F) \setminus F$. Since the right-hand side of 
	the last equation can also be written as 
	\begin{equation}
		\bigcup_{i=0}^n \underbrace{\left(\bigcirc_{j=0}^i \Phi_{R'}\right)(F)}_{Z_i}
	\end{equation}
	with $Z_i\subseteq Z_{i+1}, 0\leq i\leq {n-1}$ it follows that $x\in Z_i$ for some $i\in 1\leq {n-1}$. However, this implies
	that there is a sequence of reactions $(r_i)_{i\in\mathbb{N}}$ with $E(r_i)\in Z_{i}, P(r_i)\in Z_{i+1}, 0\leq i \leq n-1$ such
	that $\{x_i, r_i) \ | \ \SM_{(x_ir_i}<0, x_i\in E(r_i)\} \in E_1(\king')$ and $\{(r_i,x_{i+1}) \ \vert \ \SM_{x_{i+1},r_i}>0, 
	x_{i+1}\in P(r_i)\}\in E_2(\king')$. $x_0\in F$ then yields the required path.
\end{proof}
